SSH notes

Create Key

ls -al ~/.ssh  -  shows what keys you do have
ssh-keygen -C "username@domain"

chmod 700

pbcopy < ~/.ssh/id_rsa.pub
# Copies the contents of the id_rsa.pub file to your clipboard

# start the ssh-agent in the background
eval "$(ssh-agent -s)"
# Agent pid 59566
ssh-add ~/.ssh/id_rsa
